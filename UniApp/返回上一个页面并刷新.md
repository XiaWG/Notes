# uniapp返回上一页并刷新

<script>
  uni.navigateBack({
    success: () => {
      let page = getCurrentPages().pop(); //跳转页面成功之后
      if (!page) return;
      let options = {
        xx:xx // 例如上一页面是id为2的数据,要传递回去
      }
      options = page.options // page自带options对象.
      page.onLoad(options);
    }
  })
</script>
